Future<int> fetchRegister(String login, String password, String name) async {

      'https://areachad.herokuapp.com/user/register?mail=' +
          login +
          '&pass=' +
          password +
          '&name=' +
          name));

  if (response.statusCode == 200) {
    print("Good register");
    return 0;
  } else {
    print("Not good register");
    return 1;
  }
}
